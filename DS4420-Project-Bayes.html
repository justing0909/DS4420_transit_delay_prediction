use brms

# normally distributed weights
prior <- default_prior(
  travel_time_sec ~ temperature_2m...C. + wind_speed_10m..km.h. + precipitation..mm. + visibility..m. + relative_humidity_2m....,
  data = mbta_train,
  family = gaussian()
)

# define posterior
mbta_brm <- brm(
  travel_time_sec ~ temperature_2m...C. + wind_speed_10m..km.h. + precipitation..mm. + visibility..m. + relative_humidity_2m....,
  data = mbta_train,
  family = shifted_lognormal(),
  prior = prior,
  chains = 4,
  iter = 10000,
  warmup = 1800,
  cores = 1,
  file = "mbta_model_v15.rds" # 6 is good (); 14 is good
)
